import numpy as np
import sys  
from scipy import integrate
from PIL import Image
np.set_printoptions(threshold=sys.maxsize)


def aif_trapz(aif, time, timepoint = 15,Hct = 0.45):

    aif0 =  np.mean(aif[0:timepoint])
    aif_new = (aif-aif0)/(1-Hct)
    first_pass_aif_new = np.empty(np.shape(time))
    second_pass_aif_new = np.empty(np.shape(time))

    first_pass_aif_new = integrate.cumtrapz(aif_new,time)
    first_pass_aif_new = np.insert(first_pass_aif_new,0,0)#add extra zero to make array back to 265
    second_pass_aif_new = integrate.cumtrapz(first_pass_aif_new,time)
    second_pass_aif_new = np.insert(second_pass_aif_new,0,0)#add extra zero to make array back to 265
    return first_pass_aif_new, second_pass_aif_new

 
def Linear_Least_Squares_2CFM(images_to_be_fitted, time, first_pass_aif_new, second_pass_aif_new, return_parameters=True): 
    """ Linear least squares 2-compartment filtration model fit.

    Args
    ----
    ct (numpy.ndarray): pixel value (concentration) for time-series (i.e. at each dynamic measurement) with shape [x,:] 
    timepoints (list): DCE dynamic timepoints
    aif (list): arterial input function
    timepoint (int): number of baseline acquisitions
    Hct (float): hematocrit

    Returns
    -------
    fit (int): signal model fit per pixel
    Fp (numpy.float64): fitted parameter 'Fp' per pixel 
    Tp (int): fitted parameter 'Tp'
    PS (int): fitted parameter 'PS'
    Te (int): fit parameter 'Te'
    """
    shape = np.shape(images_to_be_fitted) #(147456, 265)
    S0 = np.empty(shape[0])
    St = images_to_be_fitted # signal
    Ct = np.empty(shape) #concentration
    Sfit = np.empty(shape)
    Cfit = np.empty(shape)
    
    timepoint = 15 # add user based input 
    
    for x in range(shape[0]):#pixels
      S0[x] = np.mean(St[x,0:timepoint]) # over user based: timepoint = 15 baselines only
      Ct[x,:] = St[x,:]-S0[x]
       
    time = np.tile(time, (shape[0],1)) # tile to repeat to match ct_new shape

    first_pass_ct_new = np.zeros(shape)  #(147456, 265)
    second_pass_ct_new = np.zeros(shape)  #(147456, 265)

    first_pass_ct_new  = integrate.cumtrapz(Ct,time)
    first_pass_ct_new = np.insert(first_pass_ct_new,0,0, axis=1)#add extra zero to make array back to 265

    second_pass_ct_new = integrate.cumtrapz(first_pass_ct_new,time)
    second_pass_ct_new = np.insert(second_pass_ct_new,0,0, axis=1)#add extra zero to make array back to 265
  
    A1_1 = np.array(second_pass_ct_new)
    A1_2 = np.array(first_pass_ct_new)
    A1_3 = np.array(second_pass_aif_new)
    A1_4 = np.array(first_pass_aif_new) 
    A1_3 = np.tile(A1_3, (shape[0],1)) # tile to repeat to match ct_new shape

    A1_4 = np.tile(A1_4, (shape[0],1)) # tile to repeat to match ct_new shape
    A_tuple = np.array([-A1_1, -A1_2, A1_3, A1_4])  # (4, 147456, 265)
    A = np.transpose(A_tuple,(2,0,1)) # (265, 4, 147456)

   # X = np.linalg.lstsq(A_tuple,Ct,rcond=None) # 3d array error TODO pixelwise
    X = np.empty([shape[0],4]) 
  
    for x in range(shape[0]):
        X[x,:] = np.linalg.lstsq(A[:,:,x],Ct[x,:],rcond=None)[0] #  # (265, 4)*(265)
        Cfit[x,:] =X[x,0]*A[:,0,x] + X[x,1]*A[:,1,x] + X[x,2]*A[:,2,x] + X[x,3]*A[:,3,x]
        Sfit[x,:] = S0[x]+Cfit[x,:]

    # Extract physical parameters # TODO: TBC KS
    alpha = X[:,0]
    beta = X[:,1]
    gamma = X[:,2]
    Fp = X[:,3]
    
    if alpha.all() == 0 or Fp.all() == 0 :
        print("all zeros")
        Tp = np.zeros(shape[0]) 
        PS = np.zeros(shape[0]) 
        Te = np.zeros(shape[0]) 

    else:    
        T = gamma/(alpha*Fp)  
        det = np.square(beta)-4*alpha
        if det < 0 :
            Tp = beta/(2*alpha)
            Te = beta/(2*alpha)
        else:
            Tp = (beta - np.sqrt(np.square(beta)-4*alpha))/(2*alpha)
            Te = (beta + np.sqrt(np.square(beta)-4*alpha))/(2*alpha)
        if Te == 0:
            print("te all zeros")
            Fp = np.zeros(shape[0])
            Tp = np.zeros(shape[0])
            PS = np.zeros(shape[0])

        else:    
            PS = Fp*(T-Tp)/Te                       

    # test fit result intermediate: DCE test
    fit_reshape = np.reshape(Sfit, (384, 384, 265)) # reshape the fit based on original image size
    shape_fit_reshape = np.shape(fit_reshape)
    for i in range(shape_fit_reshape[2]):
            im = Image.fromarray(fit_reshape[:,:,i])
            im.save('C:/Users/medkshaa/Documents/GitHub/MDR_Library/MDR_registration_output/DCE/intermediate_results/fit/' + str(i) + ".tiff")

    return Sfit, Fp, Tp, PS, Te 



def main(images_to_be_fitted, signal_model_parameters):
    """ main function that performs 2-compartment filtration DCE model fitting at single pixel level. 

    Args
    ----
    images_to_be_fitted (numpy.ndarray): pixel value for time-series (i.e. at each dynamic measurement) with shape [x,:]
    signal_model_parameters (list): AIF and corresponging timepoints as a list


    Returns
    -------
    fit (int): signal model fit per pixel
    fitted_parameters (list): list with signal model fitted parameters [Fp, Tp, Ps, Te]    
    """
    
    AIF = signal_model_parameters[0]
 
    timepoints = signal_model_parameters[1]
  
    first_pass_aif_new, second_pass_aif_new = aif_trapz(AIF, timepoints, timepoint = 15,Hct = 0.45) # user input for mean of baseline timepoint and Hct
  
    results = Linear_Least_Squares_2CFM(images_to_be_fitted, timepoints, first_pass_aif_new, second_pass_aif_new, return_parameters=True) 

    fit = results[0]
    Fp = results[1]
    Tp = results[2]
    Ps = results[3]
    Te = results[4]

    fitted_parameters_tuple = (Fp, Tp, Ps, Te)
    fitted_parameters = np.vstack(fitted_parameters_tuple)

    #### intermediate results save
    Fp_intermediate = fitted_parameters[0,:]
    Tp_intermediate = fitted_parameters[1,:]
    Ps_intermediate = fitted_parameters[2,:]
    Te_intermediate = fitted_parameters[3,:]

    array_Fp_intermediate = np.reshape(Fp_intermediate, [384,384]) 
    Img_Fp_intermediate = Image.fromarray(array_Fp_intermediate)
    Img_Fp_intermediate.save('C:/Users/medkshaa/Documents/GitHub/MDR_Library/MDR_registration_output/DCE/intermediate_results/fitted_params/Fp_intermediate' + ".tiff")
    
    array_Tp_intermediate = np.reshape(Tp_intermediate, [384,384]) 
    Img_Tp_intermediate = Image.fromarray(array_Tp_intermediate)
    Img_Tp_intermediate.save('C:/Users/medkshaa/Documents/GitHub/MDR_Library/MDR_registration_output/DCE/intermediate_results/fitted_params/Tp_intermediate' + ".tiff")

    array_Ps_intermediate = np.reshape(Ps_intermediate, [384,384]) 
    Img_Ps_intermediate = Image.fromarray(array_Ps_intermediate)
    Img_Ps_intermediate.save('C:/Users/medkshaa/Documents/GitHub/MDR_Library/MDR_registration_output/DCE/intermediate_results/fitted_params/Ps_intermediate' + ".tiff")

    array_Te_intermediate = np.reshape(Te_intermediate, [384,384]) 
    Img_Te_intermediate = Image.fromarray(array_Te_intermediate)
    Img_Te_intermediate.save('C:/Users/medkshaa/Documents/GitHub/MDR_Library/MDR_registration_output/DCE/intermediate_results/fitted_params/Te_intermediate' + ".tiff")
   
   # np.sys.exit()

    return fit, fitted_parameters
